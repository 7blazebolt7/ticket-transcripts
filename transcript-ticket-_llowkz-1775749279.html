(function() {
const getBoard = () => {
const cells = Array.from(document.querySelectorAll('[role="gridcell"]'));

const size = Math.sqrt(cells.length);
const board = Array(size).fill().map(() => Array(size).fill(null));

cells.forEach((cell, idx) => {
const row = Math.floor(idx / size);
const col = idx % size;
const text = cell.innerText || cell.textContent;

if (cell.classList.contains('mine') || text === '💣') board[row][col] = 'M';
else if (!isNaN(parseInt(text))) board[row][col] = parseInt(text);
else if (text === '' || cell.classList.contains('hidden')) board[row][col] = '?';
});
return { board, size, cells };
};

const getNeighbors = (r, c, size) => {
const neighbors = [];
for (let dr = -1; dr <= 1; dr++)
for (let dc = -1; dc <= 1; dc++)
if (dr !== 0 || dc !== 0) {
const nr = r + dr, nc = c + dc;
if (nr >= 0 && nr < size && nc >= 0 && nc < size)
neighbors.push([nr, nc]);
}
return neighbors;
};

const solve = () => {
const { board, size, cells } = getBoard();
let changed = false;

for (let r = 0; r < size; r++) {
for (let c = 0; c < size; c++) {
if (typeof board[r][c] === 'number') {
const neighbors = getNeighbors(r, c, size);
const unknown = neighbors.filter(([nr, nc]) => board[nr][nc] === '?');
const mines = neighbors.filter(([nr, nc]) => board[nr][nc] === 'M').length;
const remaining = board[r][c] - mines;

if (remaining === unknown.length && unknown.length > 0) {
// All unknown are mines
unknown.forEach(([ur, uc]) => {
const idx = ur * size + uc;
cells[idx]?.click(); // Flag as mine (right-click simulation)
console.log(`💣 Flag mine at (${ur},${uc})`);
});
changed = true;
}
else if (remaining === 0 && unknown.length > 0) {
// All unknown are safe
unknown.forEach(([ur, uc]) => {
const idx = ur * size + uc;
cells[idx]?.click(); // Left-click safe cell
console.log(`✅ Safe click at (${ur},${uc})`);
});
changed = true;
}
}
}
}

if (!changed) console.log('🤔 No guaranteed moves — try guessing a random cell or use hint mode');
else console.log('🎯 Moves executed!');
};
